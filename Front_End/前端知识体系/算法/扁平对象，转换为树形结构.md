实现代码

```
function toTree(obj) {
    let result = {}
    for(let key in obj) {
      let parent = obj[key].parent
      if(parent) {
        obj[parent][key] = obj[key]
      } else {
        result = obj[key]
      }
    }
    return result
}

```
